  mutate(
    TEAM_A_ABBR = abbr_map[TEAM_A],
    TEAM_B_ABBR = abbr_map[TEAM_B]
  )

df <- df %>%
  mutate(
    WINNER = factor(
      WINNER,
      levels = c(0, 1),
      labels = c("TEAM_B", "TEAM_A")
    )
  )

finals_obs <- df %>%
  filter(SERIES_GAMES == 1) %>%
  mutate(WINNER = NA,
         TEAM_A_ABBR = abbr_map[TEAM_A],
    TEAM_B_ABBR = abbr_map[TEAM_B])

df <- df %>%
  filter(SERIES_GAMES != 1)

# -----------------------------
# Train/Test Split
# -----------------------------
train_df <- df %>%
  filter(season != "2025-26")

test_df <- df %>%
  filter(season == "2025-26")

# -----------------------------
# Cross Validation
# -----------------------------
set.seed(31)

nba_folds <- vfold_cv(
  train_df,
  v = 5,
  strata = WINNER
)

# -----------------------------
# Recipe
# -----------------------------
nba_recipe <- recipe(
  WINNER ~
    WIN_PCT_DIFF +
    NET_RATING_DIFF +
    TS_DIFF +
    THREE_PT_DIFF +
    OFF_RATING_DIFF +
    DEF_RATING_DIFF +
    EFG_DIFF +
    OREB_DIFF +
    TOV_DIFF +
    PLUS_MINUS_DIFF +
    SRS_DIFF +
    H2H_TS_PCT_DIFF +
    H2H_WIN_PCT_DIFF +
    H2H_EFG_PCT_DIFF +
    H2H_POINT_DIFF_DIFF,
  data = train_df
)

# -----------------------------
# Random Forest Specification
# -----------------------------
rf_spec <- rand_forest(
  trees = 500,
  mtry = tune(),
  min_n = tune()
) %>%
  set_engine(
    "ranger",
    importance = "impurity"
  ) %>%
  set_mode("classification")

# -----------------------------
# Workflow
# -----------------------------
rf_workflow <- workflow() %>%
  add_recipe(nba_recipe) %>%
  add_model(rf_spec)

# -----------------------------
# Hyperparameter Grid
# -----------------------------
rf_grid <- expand.grid(
  mtry = c(2, 4, 6, 8, 10, 12, 15),
  min_n = c(2, 5, 10, 15, 20)
)

# -----------------------------
# Tune Model
# -----------------------------
set.seed(31)

rf_res <- tune_grid(
  rf_workflow,
  resamples = nba_folds,
  grid = rf_grid,
  metrics = metric_set(
    accuracy,
    roc_auc
  )
)

# -----------------------------
# View Results
# -----------------------------

# -----------------------------
# Finalize Workflow
# -----------------------------
final_rf <- finalize_workflow(
  rf_workflow,
  best_rf
)

# Fit Final Model
# -----------------------------

rf_preds_table <- rf_preds %>%
  mutate(
    Actual_Winner = if_else(WINNER == "TEAM_A", TEAM_A_ABBR, TEAM_B_ABBR),
    Predicted_Winner = if_else(.pred_class == "TEAM_A", TEAM_A_ABBR, TEAM_B_ABBR),
    Confidence = if_else(.pred_class == "TEAM_A", .pred_TEAM_A, .pred_TEAM_B),
    Correct = Actual_Winner == Predicted_Winner
  ) %>%
  select(
    TEAM_A = TEAM_A_ABBR,
    TEAM_B = TEAM_B_ABBR,
    Actual_Winner,
    Predicted_Winner,
    Confidence,
    Prob_TEAM_A = .pred_TEAM_A,
    Prob_TEAM_B = .pred_TEAM_B,
    Outcome = SERIES_SCORE,
    Correct
  ) %>%
  arrange(desc(Confidence))

rf_preds_table %>%
