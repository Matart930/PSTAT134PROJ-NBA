import pandas as pd
import matplotlib.pyplot as plt

# Load the CSV file into a DataFrame
nba_last_30_games = pd.read_csv('nba_last_30_games.csv')

# Convert WL to numeric values (1 for win, 0 for loss)
nba_last_30_games['is_win'] = nba_last_30_games['WL'].apply(lambda x: 1 if x == 'W' else 0)

# Calculate win rate per team
win_rate = nba_last_30_games.groupby('TEAM_NAME')['is_win'].mean().sort_values(ascending=False)

# Plotting the win rates
win_rate.plot(kind='bar', figsize=(12, 6), color='skyblue')
plt.title('Win Rate of NBA Teams in the Last 30 Games')
plt.ylabel('Win %')
plt.show()

We can see that from the last 30 games of the 2025-26 NBA regular season that the top three teams in terms of W-L record were the San Antonio Spurs, The Oklahoma City Thunder, and the Boston Celtics; all top teams that one would expect to be there. The bottom three teams in terms of W-L record were the Indiana Pacers, Memphis Grizzlies, and Washington Wizards. Now we'll take a look at other metrics, such as points per game, point differentials, and more to see if we see similar trends.

# 1. Calculate the win rate and average point differential
team_stats = nba_last_30_games.groupby('TEAM_NAME').agg({
    'WL': lambda x: (x == 'W').mean(),
    'PLUS_MINUS': 'mean'
}).reset_index()

# 2. Assign rankings (1 = highest/best)
team_stats['WL_RANK'] = team_stats['WL'].rank(ascending=False, method='min')
team_stats['PD_RANK'] = team_stats['PLUS_MINUS'].rank(ascending=False, method='min')

# 3. Calculate Difference: (PD Rank - WL Rank)
# If PD Rank is 3rd and WL Rank is 1st: 3 - 1 = +2 (Overperforming)
team_stats['RANK_DIFF'] = team_stats['PD_RANK'] - team_stats['WL_RANK']

# 4. Sort by WL Record, and PD Rank for ties
team_stats = team_stats.sort_values(by=['WL_RANK', 'PD_RANK'])

# 5. Plotting
fig, ax = plt.subplots(figsize=(15, 9))
colors = ['#2ca02c' if x >= 0 else '#d62728' for x in team_stats['RANK_DIFF']]
bars = ax.bar(team_stats['TEAM_NAME'], team_stats['RANK_DIFF'], color=colors)

# 6. Difference Labels (on the bars)
ax.bar_label(bars, padding=3, fontweight='bold')

# 7. WL Rank Labels (Placed lower down)
# We calculate a 'bottom' value based on the lowest bar to ensure it's below everything
y_offset = team_stats['RANK_DIFF'].min() - 8 

for i, row in enumerate(team_stats.itertuples()):
    ax.text(i, y_offset, f"#{row.WL_RANK}", 
            ha='center', va='top', fontsize=9, 
            fontweight='bold', color='white',
            bbox=dict(facecolor='black', alpha=0.6, boxstyle='round,pad=0.3'))

# Styling
plt.xticks(rotation=90, fontsize=10)
plt.axhline(0, color='black', linewidth=1)
plt.title('NBA Performance: Record vs. Point Differential\n(Black labels = WL Rank | Bar Height = Luck/Overperformance)', fontsize=14)
plt.ylabel('Ranking Difference', fontsize=12)

# Adjusting margins to make room for the new lower labels
plt.subplots_adjust(bottom=0.25) 
plt.grid(axis='y', linestyle='--', alpha=0.3)

plt.show()
